  dist.init_process_group('nccl')
  from mamformer.parallelism import ExpertParallelGroup, EPMoE
  from mamformer.parallelism.tensor_parallel import set_comm_diagnostics

              routed_expert_dim=128, ep_group=ep).cuda()

  print(f'Rank {dist.get_rank()}: output shape={out.shape}, local experts={info[\"local_experts\"]}')
  # 兩張卡輸出不同的 local_experts 數量 = EP 正確運作

  完整驗證清單

  # 1. 單卡基準
  python scripts/train.py --config configs/debug.yaml --bf16

  # 2. 雙卡 EP（各持一半專家）
  torchrun --nproc_per_node=2 scripts/train_distributed.py \
      --config configs/ultra-7b.yaml --ep 2 --bf16

  # 3. 雙卡 TP（單層切兩半）
  torchrun --nproc_per_node=2 scripts/train_distributed.py \
      --config configs/ultra-7b.yaml --tp 2 --bf16

  # 4. 雙卡 PP（層切兩段）
  torchrun --nproc_per_node=2 scripts/train_distributed.py \
      --config configs/ultra-7b.yaml --pp 2 --bf16

  # 5. 四卡全開
  torchrun --nproc_per_node=4 scripts/train_distributed.py \
      --config configs/ultra-7b.yaml --tp 2 --pp 2 --bf16
